在Linux常用的压缩格式如下表：

格式 压缩工具
.zip zip压缩工具（*）
.gz gzip压缩工具，只能压缩文件，会删除原文件(通常配合tar使用)
.bz2 bzip2压缩工具，只能压缩文件，会删除原文件(通常配合tar使用)
.tar.gz 先使用tar命令归档打包，然后使用gzip压缩 (*)
.tar.bz2 先使用tar命令归档打包，然后使用bzip压缩 (*)

gzip

gzip命令

打包与压缩，仅对文件有效。如新建一个文件wu.txt，文件内容为Hello。

[root@localhost ~]# echo "Hello" > wu.txt

把文件打包，源文件会消失。

[root@localhost ~]# yum install gzip
[root@localhost ~]# gzip wu.txt 
[root@localhost ~]# ll
total 4
-rw-r--r--. 1 root root 33 Aug 13 20:38 wu.txt.gz

可以通过zcat查看压缩文件内容

[root@localhost ~]# zcat wu.txt.gz 
Hello

-d选项可为文件解压，解压后压缩包消失。

[root@localhost ~]# gzip -d wu.txt.gz 
[root@localhost ~]# ll
total 4
-rw-r--r--. 1 root root 6 Aug 13 20:38 wu.txt

应用场景，让某个文件快速关闭和启用。

[root@localhost yum.repos.d]# gzip CentOS-Vault.repo 
[root@localhost yum.repos.d]# gzip -d CentOS-Vault.repo

zip

zip命令

需要安装zip相关工具

[root@localhost ~]# yum install zip unzip

压缩文件

[root@localhost ~]# zip test.zip test.txt 
  adding: test.txt (deflated 56%)
[root@localhost ~]# ll
total 8
-rw-r--r--. 1 root root 979 Aug 13 10:20 test.txt
-rw-r--r--. 1 root root 599 Aug 13 10:22 test.zip

递归压缩，把dir目录和wu.txt文件都压缩到dir.zip中

[root@localhost ~]# zip -r dir.zip dir/ wu.txt 
  adding: dir/ (stored 0%)
  adding: dir/wu.txt (stored 0%)
  adding: dir/test/ (stored 0%)
  adding: dir/hello.txt (deflated 56%)
  adding: wu.txt (stored 0%)

查看包内容

[root@localhost ~]# unzip -l dir.zip 
Archive:  dir.zip
  Length      Date    Time    Name
---------  ---------- -----   ----
        0  08-13-2019 20:58   dir/
        7  08-13-2019 20:57   dir/wu.txt
        0  08-13-2019 20:57   dir/test/
      979  08-13-2019 20:57   dir/hello.txt
        6  08-13-2019 20:38   wu.txt
---------                     -------
      992                     5 files

解压到当前目录

[root@localhost ~]# unzip dir.zip 
Archive:  dir.zip
   creating: dir/
 extracting: dir/wu.txt              
   creating: dir/test/
  inflating: dir/hello.txt           
 extracting: wu.txt

解压到指定目录下

[root@localhost ~]# unzip dir.zip -d /opt/
Archive:  dir.zip
   creating: /opt/dir/
 extracting: /opt/dir/wu.txt         
   creating: /opt/dir/test/
  inflating: /opt/dir/hello.txt      
 extracting: /opt/wu.txt

tar

压缩

# 以gzip的方式压缩文件
[root@localhost ~]# tar czf test.tar.gz dir dir.txt 
# 以bz2的方式压缩文件
[root@localhost ~]# tar cjf test.tar.bz2 test test2 dir.txt

查看压缩文件

# 不论查看何种类型的压缩文件都是用tf选项
[root@localhost ~]# tar tf test.tar.gz 
dir/
dir.txt
[root@localhost ~]# tar tf test.tar.bz2 
test/
test2/
dir.txt

解压文件

# 不论解压何种类型的文件，都使用xf选项
[root@localhost ~]# tar xf test.tar.gz 
[root@localhost ~]# tar xf test.tar.bz2
# -C可指定解压的目录 
[root@localhost ~]# tar xf test.tar.gz -C /opt

打包/tmp下所有文件

[root@localhost ~]# find /var/log -type f | xargs tar czf log.tar.gz
tar: Removing leading `/' from member names
[root@localhost ~]# tar tf log.tar.gz 
var/log/tuned/tuned.log
var/log/audit/audit.log
var/log/cron
var/log/messages
var/log/yum.log
var/log/boot.log
var/log/wtmp

压缩时，排除文件

# 压缩/etc排除单个文件
[root@localhost ~]# tar czf etc.tar.gz /etc/ --exclude=etc/services
# 压缩/etc排除多个文件
[root@localhost ~]# tar czf etc.tar.gz /etc/ --exclude=etc/services --exclude=etc/shadow
# 将要排除的文件写入文本，通过X选项排除
[root@localhost ~]# cat hello.txt 
/etc/passwd
/etc/passwd-
/etc/locale.conf
/etc/hostname
/etc/.updated
/etc/aliases.db
/etc/vimrc
/etc/wgetrc
[root@localhost ~]# tar czXf hello.txt etc.tar.gz /etc/
tar: Removing leading `/' from member names
